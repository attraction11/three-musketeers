    font-size: 14px;
    animation: fontSize 5s infinite alternate ease-in-out;
}
p > img {
    width: 16px; height: 16px;
    /* 下面的ex实现是为了方便大家对比效果 */
    /* vertical-align: .6ex; */
    vertical-align: 25%;
    position: relative;
    top: 8px;
}
@keyframes fontSize {
    to {
        font-size: 60px;
    }
}
